<wiring_patch>
.module = "github.com/example/my-app"
| .language = "python"
| .packages["src/api"].owns = ["CreateOrder", "ListOrders"]
| .packages["src/services"].owns = ["OrderService"]
| .symbols["src/services.OrderService"] = {{"package":"src/services","signature":"class OrderService","exports":["OrderService"]}}
| .deps += [{{"from":"src/api","to":"src/services"}}]

Rules: jq only; symbol keys are "package.Name"; .module is the local import root
(not bare "cmd"/"src"/"api"/"service" — use a real package name like
"github.com/org/app", "@org/app", or "my_app"); one package per concern;
